## 2. HTTP 客户端封装

### 2.1 客户端代码

```python
# app/clients/__init__.py
from app.clients.account_client import AccountClient, InsufficientBalanceError

# app/clients/account_client.py
"""
账户服务 HTTP 客户端。

用法：
    from app.clients import AccountClient
    client = AccountClient()
    balance = client.get_balance(user_id)
    result = client.transfer(from_user, to_user, amount, idempotency_key)

所有方法都是同步的（适配 Flask 同步模型）。
底层使用 httpx 同步 Client。
"""
import hmac
import hashlib
import json
import time
import httpx
from decimal import Decimal
from flask import current_app


class AccountClientError(Exception):
    """账户服务通用错误"""
    def __init__(self, message, code=None, detail=None):
        self.message = message
        self.code = code
        self.detail = detail
        super().__init__(message)


class InsufficientBalanceError(AccountClientError):
    """余额不足"""
    def __init__(self, user_id, required, available):
        self.user_id = user_id
        self.required = required
        self.available = available
        super().__init__(
            f"用户 {user_id} 小鱼干不足：需要 {required}，可用 {available}",
            code=400,
            detail={"user_id": user_id, "required": required, "available": available},
        )


class AccountClient:
    """同步 HTTP 客户端，封装对账户服务的调用。

    核心设计：
    - 博客系统本身是账户服务中的一个账户（user_id = 'raricy-blog-system'）
    - 博客持有自己的 API Key（存在 .env 中）
    - 博客替用户保管他们的 API Key（加密存储在 users 表）
    - 转账时，根据 from_user_id 选择正确的 Key：
      - from 是博客系统 → 用系统 Key
      - from 是普通用户 → 从 DB 取出用户加密的 Key，解密后使用
    """

    SYSTEM_USER_ID = "raricy-blog-system"

    def __init__(self, app=None):
        self.base_url = None
        self.system_key = None     # 博客系统自己的 API Key（从 .env 读）
        self.internal_token = None # X-Internal-Token 共享密钥（从 .env 读）
        self.timeout = 5.0
        self._cipher = None        # Fernet 加密器（用于加/解密用户 Key）
        if app:
            self.init_app(app)

    def init_app(self, app):
        import base64
        from cryptography.fernet import Fernet

        self.base_url = app.config.get("ACCOUNT_SERVICE_URL", "http://localhost:8000")
        self.system_key = app.config.get("ACCOUNT_SYSTEM_KEY", "")
        self.internal_token = app.config.get("ACCOUNT_SERVICE_INTERNAL_TOKEN", "")
        self.timeout = app.config.get("ACCOUNT_SERVICE_TIMEOUT", 5.0)

        # 从 SECRET_KEY 派生 Fernet 密钥（用于加密用户 API Key）
        derived = base64.urlsafe_b64encode(
            hashlib.sha256(app.config["SECRET_KEY"].encode()).digest()
        )
        self._cipher = Fernet(derived)

    # ── API Key 管理 ───────────────────────────────

    def _encrypt_key(self, plain_key: str) -> str:
        """加密用户 API Key（存 DB 用）。"""
        return self._cipher.encrypt(plain_key.encode()).decode()

    def _decrypt_key(self, encrypted: str) -> str:
        """解密用户 API Key（调 API 时用）。"""
        return self._cipher.decrypt(encrypted.encode()).decode()

    def _get_api_key_for(self, user_id: str) -> str:
        """根据 user_id 获取对应的 API Key。

        - 博客系统 → 返回 .env 中的系统 Key
        - 普通用户 → 从 DB 解密用户的 Key
        """
        if user_id == self.SYSTEM_USER_ID:
            return self.system_key

        from app.models.user import User
        user = User.query.get(user_id)
        if not user or not user.fish_api_key_encrypted:
            raise AccountClientError(
                f"用户 {user_id} 没有关联的鱼干账户 API Key",
                code=400,
            )
        return self._decrypt_key(user.fish_api_key_encrypted)

    # ── 内部方法 ─────────────────────────────────

    def _request(
        self, method: str, path: str,
        json_data: dict = None,
        idempotency_key: str = None,
        api_key: str = None,   # 可选的 API Key（余额查询等不需要）
    ):
        """发送请求到账户服务。所有请求（除 /health）需 X-Internal-Token。"""
        url = f"{self.base_url}{path}"
        body = json.dumps(json_data) if json_data else ""

        headers = {
            "Content-Type": "application/json",
            "X-Internal-Token": self.internal_token,  # 所有请求携带
        }
        if api_key:
            headers["Authorization"] = f"Bearer {api_key}"
        if idempotency_key:
            headers["X-Idempotency-Key"] = idempotency_key

        try:
            with httpx.Client(timeout=self.timeout) as http:
                resp = http.request(
                    method, url,
                    content=body if body else None,
                    headers=headers,
                )
                data = resp.json()

                if resp.status_code >= 400:
                    self._handle_error(resp.status_code, data)

                return data.get("data", data)
        except httpx.RequestError as e:
            raise AccountClientError(f"账户服务不可达: {e}", code=503)
        except httpx.TimeoutException:
            raise AccountClientError("账户服务超时", code=503)

    def _handle_error(self, status_code: int, data: dict):
        message = data.get("message", "未知错误")
        detail = data.get("detail")
        code = data.get("code", status_code)

        if "小鱼干不足" in message or "余额不足" in message:
            raise InsufficientBalanceError(
                user_id=detail.get("user_id", ""),
                required=detail.get("required", 0),
                available=detail.get("available", 0),
            )
        raise AccountClientError(message, code=code, detail=detail)

    # ── 公开 API ─────────────────────────────────

    def create_account(self, user_id: str, currency: str = "DRIED_FISH") -> dict:
        """用户注册时创建账户。返回的 api_key 加密存储到用户表。
        幂等：重复调用不报错，但也不返回 Key。"""
        result = self._request(
            "POST", "/api/v1/accounts",
            json_data={"user_id": user_id, "currency": currency},
        )

        # 如果是新建账户，api_key 字段存在；加密存储
        api_key = result.get("api_key")
        if api_key:
            from app.models.user import User
            from app.extensions import db
            user = User.query.get(user_id)
            if user:
                user.fish_api_key_encrypted = self._encrypt_key(api_key)
                db.session.commit()

        return result

    def get_balance(self, user_id: str) -> float:
        """查询单个用户余额。不存在的用户返回 0.0。无需认证。"""
        data = self._request("GET", f"/api/v1/accounts/{user_id}/balance")
        return float(data["balance"])

    def get_balances(self, user_ids: list[str]) -> dict[str, float]:
        """批量查询余额。无需认证。"""
        data = self._request(
            "POST", "/api/v1/accounts/balances/batch",
            json_data={"user_ids": user_ids},
        )
        return {uid: float(bal) for uid, bal in data.get("balances", {}).items()}

    def transfer(
        self,
        from_user_id: str,
        to_user_id: str,
        amount: float,
        entry_type: str,
        description: str = "",
        metadata: dict = None,
        idempotency_key: str = None,
    ) -> dict:
        """转账。自动根据 from_user_id 选择正确的 API Key。

        - from_user_id = "raricy-blog-system" → 用系统 Key（签到发鱼干等）
        - from_user_id = 普通用户 → 从 DB 解密该用户的 Key（投喂等）

        抛出 InsufficientBalanceError 如果余额不足。
        抛出 AccountClientError (401) 如果 API Key 无效。
        抛出 AccountClientError (403) 如果 Key 与 from_user_id 不匹配。
        """
        api_key = self._get_api_key_for(from_user_id)

        return self._request(
            "POST", "/api/v1/transfers",
            json_data={
                "from_user_id": from_user_id,
                "to_user_id": to_user_id,
                "amount": amount,
                "entry_type": entry_type,
                "description": description,
                "metadata": metadata or {},
            },
            idempotency_key=idempotency_key,
            api_key=api_key,  # ← Bearer token
        )

    def get_ledger(
        self, user_id: str, page: int = 1, per_page: int = 20,
        entry_type: str = None, start: str = None, end: str = None,
    ) -> dict:
        """查询交易流水。"""
        params = {"page": page, "per_page": per_page}
        if entry_type:
            params["entry_type"] = entry_type
        if start:
            params["start"] = start
        if end:
            params["end"] = end

        query_string = "&".join(f"{k}={v}" for k, v in params.items())
        return self._request(
            "GET", f"/api/v1/accounts/{user_id}/ledger?{query_string}",
        )
```

```python
from flask import current_app
import uuid

def feed_fish(blog_id, user_id, count):
    blog = Blog.query.get(blog_id)
    if not blog or blog.ignore:
        raise ValueError('文章不存在')

    # 1. 博客侧：验证投喂上限（业务逻辑保留在博客）
    existing = BlogFeed.query.filter_by(blog_id=blog_id, user_id=user_id).first()
    if existing and existing.amount + count > 5:
        raise ValueError('投喂已满（单篇文章每人最多投喂 5 条）')

    # 2. 用户支出（80% 消耗，20% 给作者）
    author_income = count * 0.2  # 现在是自然单位

    client = current_app.account_client
    idem_key = f"feed-{blog_id}-{user_id}-{count}-{uuid.uuid4().hex[:8]}"

    # 用户支出 → 系统回收 80%（用用户的 Key）
    client.transfer(
        from_user_id=user_id,
        to_user_id=AccountClient.SYSTEM_USER_ID,
        amount=count * 0.8,
        entry_type="feed_consume",
        description=f"投喂文章「{blog.title}」消耗",
        metadata={"blog_id": blog_id},
        idempotency_key=idem_key + "-consume",
    )

    # 用户支出 → 作者获得 20%（用用户的 Key）
    client.transfer(
        from_user_id=user_id,
        to_user_id=blog.author_id,
        amount=author_income,
        entry_type="feed_income",
        description=f"投喂文章「{blog.title}」分成",
        metadata={"blog_id": blog_id},
        idempotency_key=idem_key + "-income",
    )

    # 3. 博客侧：更新投喂记录（业务数据）
    if existing:
        existing.amount += count
    else:
        db.session.add(BlogFeed(blog_id=blog_id, user_id=user_id, amount=count))

    # 4. 更新文章鱼干计数
    blog.fish_count = (blog.fish_count or 0) + count
    db.session.commit()

    # 5. 通知作者（保持不变）
    if user_id != blog.author_id:
        send_notification(...)
```

> **注意**：`BlogFeed` 表保留在博客中，因为它存储的是"投喂业务规则"（每篇文章每人最多 5 条），这是博客域的逻辑，不属于账户服务。账户服务只知道"A 转钱给 B"。

```python
@auth_bp.route('/fish')
@login_required
def fish_balance():
    client = current_app.account_client
    balance = client.get_balance(current_user.id)
    # 今日获得需要通过流水查询（或缓存）
    ledger = client.get_ledger(
        current_user.id, page=1, per_page=50,
        entry_type="checkin",

        'auth/fish.html',

        today_fish=today_fish * 10000,

@auth_bp.route('/fish/transactions')
@login_required
def fish_transactions():
    page = request.args.get('page', 1, type=int)
    type_filter = request.args.get('type', None)
    client = current_app.account_client
    data = client.get_ledger(
        current_user.id, page=page, per_page=20,
        entry_type=type_filter,
    )
    # 适配模板：将 AccountClient 返回格式转为旧模板格式
    return render_template(
        'auth/fish_transactions.html',
        transactions=[_adapt_tx(tx) for tx in data['entries']],
        pagination={
            'page': data['pagination']['page'],
            'pages': data['pagination']['pages'],
            'total': data['pagination']['total'],
            'has_prev': data['pagination']['has_prev'],
            'has_next': data['pagination']['has_next'],
            'prev_num': data['pagination']['page'] - 1 if data['pagination']['has_prev'] else None,
            'next_num': data['pagination']['page'] + 1 if data['pagination']['has_next'] else None,
        },
        type_filter=type_filter,
    )

def _adapt_tx(entry: dict) -> dict:
    """将账户服务返回的 entry 转为旧 FishTransaction.to_dict() 格式"""
    return {
        'id': entry['id'],
        'amount': int(entry['amount'] * 10000),  # 转回旧内部单位
        'type': entry['entry_type'],              # entry_type → type
        'description': entry['description'],
        'created_at': entry['created_at'],
    }
```

> **过渡策略**：在 Phase 1-2，模板仍然期望旧格式（`amount` 为内部整型）。用 `_adapt_tx()` 做一层转换，前端模板零改动。Phase 3 可以重构模板直接使用新格式。

## 4. 兼容过渡：双写模式

在 Phase 1（账户服务搭建完成但不想立刻切流量），可以**双写**：

```python
# 包装函数：同时写本地和远程，但对账以远程为准
def _transitional_add_fish(user_id, amount_internal, type, description, **kwargs):
    """过渡期函数：同时写本地 DB 和远程账户服务"""
    # 1. 写本地（旧逻辑）
    add_fish(user_id, amount_internal, type, description, **kwargs)

    # 2. 写远程（新逻辑）
    try:
        current_app.account_client.transfer(
            from_user_id=AccountClient.SYSTEM_USER_ID,
            to_user_id=user_id,
            amount=amount_internal / FISH_UNIT,  # 内部→外部
            entry_type=type,
            description=description,
            idempotency_key=f"migration-{type}-{user_id}-{int(time.time())}",
        )
    except Exception as e:
        # 只记日志，不阻断主流程（过渡期本地为准）
        current_app.logger.warning(f"账户服务同步失败: {e}")
```
